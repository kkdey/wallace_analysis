Extracting enriched birds

topics <- topics_clust[[10]]
freq <- topics$freq
which_comp <- apply(freq, 1, function(x) return(which.max(x)))
freq2 <- freq[which(which_comp==1),]
idx <- order(freq2[,1])
bird_in_10_names <- rownames(freq2)[idx]
